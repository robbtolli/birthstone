CODE     ->    BLOCK CODE | STMT CODE | e
BLOCK    ->    { CODE }
STMT     ->		EXPR ;
EXPR     ->    TERM EXPR2
EXPR2		->		OP1 EXPR | e		 
TERM     ->    FACTOR | TERM2
TERM2		->		OP0 TERM | e
FACTOR   ->    str | bool | num | id | '(' EXPR ')'
OP0		->		* | / 
OP1		->		+ | -
OP2		->		< | <= | == | != | >= | > 
OP3		->		&&
OP4		->		||
OP5		->		= | := | +=
